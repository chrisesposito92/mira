    **CONN-06 Auto-suggest logic (D-11):**
    Use a `$effect` that watches `sourceId` and `targetId`:
    ```typescript
    $effect(() => {
        if (!sourceId || !targetId) return;
        const sourceSystem = systems.find(s => s.id === sourceId);
        const targetSystem = systems.find(s => s.id === targetId);
        if (!sourceSystem || !targetSystem) return;

        const isSourceHub = sourceSystem.role === 'hub';
        const isTargetHub = targetSystem.role === 'hub';

        if (isSourceHub || isTargetHub) {
            const otherSystem = isSourceHub ? targetSystem : sourceSystem;
            // Look up in component library to check is_native_connector
            const libItem = componentLibrary.find(
                c => c.id === otherSystem.component_library_id
            );
            if (libItem?.is_native_connector) {
                connectionType = 'native_connector';
            }
        }
    });
    ```

    **Validation (per validation rules in UI-SPEC):**
    ```typescript
    let errors = $state<Record<string, string>>({});

    function validate(): boolean {
        errors = {};
        if (!sourceId) errors.source = 'Select a source system';
        if (!targetId) errors.target = 'Select a target system';
        if (sourceId && targetId && sourceId === targetId) {
            errors.target = 'Source and target must be different systems';
        }
        if (label.length > 60) errors.label = 'Label must be 60 characters or fewer';
        // Check duplicate: same source+target pair (ignoring direction)
        const existing = diagramStore.currentDiagram?.content.connections ?? [];
        const isDuplicate = existing.some(c =>
            c.id !== editingConnection?.id &&
            ((c.source_id === sourceId && c.target_id === targetId) ||
             (c.source_id === targetId && c.target_id === sourceId))
        );
        if (isDuplicate) errors.duplicate = 'This connection already exists';
        return Object.keys(errors).length === 0;
    }
    ```

    **Submit handler:**
    ```typescript
    function handleSubmit() {
        if (!validate()) return;
        const connection: DiagramConnection = {
            id: editingConnection?.id ?? crypto.randomUUID(),
            source_id: sourceId,
            target_id: targetId,
            label,
            direction,
            connection_type: connectionType,
        };
        onsubmit(connection);
    }
    ```
