    CRITICAL DISABLED TOOLTIP PATTERN (addresses Codex MEDIUM concern): Disabled buttons do not emit hover/focus events in most browsers, so wrapping a disabled `<Button>` directly in `Tooltip.Trigger` will not show the tooltip. The fix is to wrap the disabled button in an inline `<span>` element that receives the hover/focus events. The `<span>` acts as the tooltip trigger, and the `<Button>` inside it is purely visual.

    When `disabled` is true:
    ```svelte
    <Tooltip.Provider>
      <Tooltip.Root>
        <Tooltip.Trigger asChild>
          <span class="inline-flex">
            <Button variant="outline" size="sm" disabled>
              <Download class="mr-2 size-4" />
              Export
            </Button>
          </span>
        </Tooltip.Trigger>
        <Tooltip.Content>
          <p>Add at least one system to export</p>
        </Tooltip.Content>
      </Tooltip.Root>
    </Tooltip.Provider>
    ```

    The `<span class="inline-flex">` wrapper is the key pattern: it receives pointer events (hover, focus) even though the child button is disabled. The `asChild` prop on `Tooltip.Trigger` may be needed to pass trigger behavior through -- check the shadcn-svelte tooltip trigger implementation. If `asChild` is not supported by this version of shadcn-svelte/bits-ui, use the `{#snippet child({ props })}` pattern instead and spread props onto the `<span>`.

    import { describe, it, expect, vi } from 'vitest';

    // Test the logic patterns used by ExportDropdown
    describe('ExportDropdown logic', () => {
      describe('handleExport re-entry guard', () => {
        it('prevents concurrent exports', async () => {
          let exporting = false;
          const mockExport = vi.fn().mockResolvedValue(undefined);

          async function handleExport() {
            if (exporting) return;
            exporting = true;
            try {
              await mockExport();
            } finally {
              exporting = false;
            }
          }

          // Simulate double-click: call twice without awaiting
          const p1 = handleExport();
          const p2 = handleExport(); // should be blocked by guard
          await Promise.all([p1, p2]);

          expect(mockExport).toHaveBeenCalledTimes(1);
        });
      });

      describe('error handling', () => {
        it('calls toast.error when export throws', async () => {
          // Test pattern: exportDiagram throws -> toast.error is called
          const errors: string[] = [];
          const mockToastError = (msg: string) => errors.push(msg);
          const mockExport = vi.fn().mockRejectedValue(new Error('Canvas fail'));

          try {
            await mockExport();
          } catch {
            mockToastError('Export failed. Try refreshing the page and exporting again.');
          }

          expect(errors).toContain('Export failed. Try refreshing the page and exporting again.');
        });
      });

      describe('hasUserSystems check', () => {
        it('returns false when only hub and prospect systems exist', () => {
          const systems = [
            { id: '1', name: 'm3ter', role: 'hub' as const },
            { id: '2', name: 'Your Product', role: 'prospect' as const },
          ];
          const hasUserSystems = systems.filter(
            (s) => s.role !== 'hub' && s.role !== 'prospect',
          ).length > 0;
          expect(hasUserSystems).toBe(false);
        });

        it('returns true when at least one non-hub non-prospect system exists', () => {
          const systems = [
            { id: '1', name: 'm3ter', role: 'hub' as const },
            { id: '2', name: 'Your Product', role: 'prospect' as const },
            { id: '3', name: 'Salesforce', role: 'system' as const },
          ];
          const hasUserSystems = systems.filter(
            (s) => s.role !== 'hub' && s.role !== 'prospect',
          ).length > 0;
          expect(hasUserSystems).toBe(true);
        });

        it('returns true for system with null role', () => {
          const systems = [
            { id: '1', name: 'm3ter', role: 'hub' as const },
            { id: '2', name: 'Custom System', role: null },
          ];
          const hasUserSystems = systems.filter(
            (s) => s.role !== 'hub' && s.role !== 'prospect',
          ).length > 0;
