    ```typescript
    async function handleExport(format: ExportFormat) {
      try {
        await exportDiagram(format, { customerName, title });
      } catch (err) {
        // Check if this is a font-only failure (export still completed)
        // The exportDiagram function handles font failure internally with console.warn
        // and continues the export. If we reach the catch block, it's a real failure.
        toast.error('Export failed. Try refreshing the page and exporting again.');
      }
    }
    ```

    - The DropdownMenu is NOT rendered when disabled -- only the tooltip-wrapped button
