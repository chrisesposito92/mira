```python
"""Seed logo images for component library entries.

Fetches logos from Logo.dev for all component_library rows where logo_base64 IS NULL.
Falls back to monogram marker when Logo.dev returns no result (per D-08).

Usage:
    cd backend && source .venv/bin/activate
    python -m scripts.seed_logos

Requires LOGO_DEV_TOKEN in backend/.env
"""

import asyncio
import base64
import hashlib
import logging

import httpx

from app.config import settings
from app.db.client import get_supabase

logging.basicConfig(level=logging.INFO, format="%(levelname)s: %(message)s")
logger = logging.getLogger(__name__)

# Monogram color palette (deterministic by name hash, per D-08)
MONOGRAM_COLORS = [
    "#4A90D9", "#50C878", "#E67E22", "#9B59B6",
    "#E74C3C", "#1ABC9C", "#F39C12", "#3498DB",
]


def _monogram_for(name: str) -> str:
    """Generate a monogram marker string.

    Format: 'monogram:<INITIALS>:<COLOR>'
    e.g., 'monogram:SF:#4A90D9' for Salesforce
    """
    words = name.split()
    if len(words) >= 2:
        initials = (words[0][0] + words[1][0]).upper()
    else:
        initials = name[:2].upper()
    color_idx = int(hashlib.md5(name.encode()).hexdigest(), 16) % len(MONOGRAM_COLORS)
    color = MONOGRAM_COLORS[color_idx]
    return f"monogram:{initials}:{color}"


async def fetch_logo(client: httpx.AsyncClient, domain: str, token: str) -> str | None:
    """Fetch a logo from Logo.dev, return base64 data URL or None."""
    url = f"https://img.logo.dev/{domain}?token={token}&size=128&format=png"
    try:
        resp = await client.get(url, follow_redirects=True, timeout=10.0)
        if resp.status_code == 200 and len(resp.content) > 100:
            b64 = base64.b64encode(resp.content).decode()
            content_type = resp.headers.get("content-type", "image/png")
            return f"data:{content_type};base64,{b64}"
    except httpx.RequestError as e:
        logger.warning("Request error for %s: %s", domain, e)
    return None


async def main():
    token = settings.logo_dev_token
    if not token:
        logger.error("LOGO_DEV_TOKEN not configured in backend/.env. Generating monograms only.")

    supabase = get_supabase()

    # Fetch all rows where logo_base64 is null
    result = supabase.table("component_library").select("id, name, domain, logo_base64").execute()
    rows = [r for r in result.data if not r.get("logo_base64")]

    if not rows:
        logger.info("All component library entries already have logos. Nothing to do.")
        return

    logger.info("Found %d entries without logos.", len(rows))

    fetched = 0
    monogrammed = 0

    async with httpx.AsyncClient() as client:
        for row in rows:
            logo_b64 = None
            if token and row["domain"]:
                logo_b64 = await fetch_logo(client, row["domain"], token)
                if logo_b64:
                    fetched += 1

            if not logo_b64:
                logo_b64 = _monogram_for(row["name"])
                monogrammed += 1

            supabase.table("component_library").update(
                {"logo_base64": logo_b64}
            ).eq("id", row["id"]).execute()

            logger.info(
                "  %s: %s",
                row["name"],
                "fetched" if not logo_b64.startswith("monogram:") else "monogram",
            )

            # Rate limit: Logo.dev recommends modest pace
            if token:
                await asyncio.sleep(0.3)

    logger.info("Done: %d logos fetched, %d monograms generated.", fetched, monogrammed)


if __name__ == "__main__":
    asyncio.run(main())
```

Key design points:
- Uses `get_supabase()` from `app.db.client` (service-role client, same as existing scripts)
- Gracefully handles missing LOGO_DEV_TOKEN by falling back to monograms for all entries
- Monogram format: `monogram:<INITIALS>:<COLOR>` -- Phase 2 renderer parses this to generate SVG monograms
- Rate-limits at 0.3s between requests to be polite to Logo.dev API
- Only processes rows where `logo_base64 IS NULL` (idempotent -- safe to re-run)
