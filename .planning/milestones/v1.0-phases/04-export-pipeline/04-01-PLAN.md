      ```typescript
      export function arrayBufferToBase64(buffer: ArrayBuffer): string {
        const bytes = new Uint8Array(buffer);
        let binary = '';
        for (let i = 0; i < bytes.byteLength; i++) {
          binary += String.fromCharCode(bytes[i]);
        }
        return btoa(binary);
      ```

    **`svgToDataUrl(svgString: string): string`** (addresses Codex HIGH concern -- dedicated UTF-8-safe helper):
    This is a NAMED, TESTED helper function that safely converts an SVG string containing potential Unicode characters (customer names, labels with accented chars, etc.) into a base64 data URL for use as an Image src. The `btoa(unescape(encodeURIComponent(...)))` pattern handles the Unicode-to-Latin1 conversion.

    ```typescript

      const base64 = btoa(unescape(encodeURIComponent(svgString)));
      return `data:image/svg+xml;base64,${base64}`;
    }
    ```

    **`validateImageDataUrls(svgDoc: Document): boolean`** (addresses both reviewers' MEDIUM concern -- EXPO-03 explicit assertion):
    Traverses all `<image>` elements in a parsed SVG document and verifies every `href` (or `xlink:href`) starts with `data:`. This is the explicit pre-export assertion that logos are base64-inlined, not external URLs.

    ```typescript
    export function validateImageDataUrls(svgDoc: Document): boolean {
      const images = svgDoc.querySelectorAll('image');
      for (const img of images) {
        const href = img.getAttribute('href') || img.getAttributeNS('http://www.w3.org/1999/xlink', 'href');
        if (href && !href.startsWith('data:')) {
          return false;
        }
      }
      return true;
    }
    ```

    **`injectFont(svgDoc: Document, fontBase64: string): void`** (DOM-based, addresses Codex HIGH and Gemini MEDIUM -- replaces regex string surgery):
    Uses DOM manipulation on a cloned SVG Document. Creates a `<style>` element and prepends it as the first child of the root `<svg>` element. Mutates the document in place (it's already a clone).

    ```typescript
    export function injectFont(svgDoc: Document, fontBase64: string): void {
      const styleEl = svgDoc.createElementNS('http://www.w3.org/2000/svg', 'style');
      styleEl.textContent = `
        @font-face {
          font-family: 'Inter';
          font-style: normal;
          font-display: swap;
          font-weight: 100 900;
          src: url(data:font/woff2;base64,${fontBase64}) format('woff2');
        }
      `;
      const svgRoot = svgDoc.documentElement;
      svgRoot.insertBefore(styleEl, svgRoot.firstChild);
    }
    ```

    **`fixContextStroke(svgDoc: Document): void`** (DOM-based, addresses Codex HIGH and Gemini MEDIUM -- replaces regex string surgery):
    Uses DOM manipulation on a cloned SVG Document. The approach:

    1. Find the `<defs>` element in the SVG document.
    2. Find all `<line>` elements and collect their unique `stroke` attribute values (hex colors from CONNECTION_COLORS).
    3. Find the original `arrowhead` and `source-dot` marker elements in `<defs>`.
    4. For each unique stroke color, clone the original markers, set a new ID (e.g., `arrowhead-00C853`), and replace `fill="context-stroke"` with the explicit hex color on the child elements (`<path>` and `<circle>`).
    5. Append new markers to `<defs>`.
    6. For each `<line>` element, read its `stroke`, and rewrite its `marker-start` and `marker-end` attributes to reference the color-specific marker IDs.

    ```typescript
    export function fixContextStroke(svgDoc: Document): void {
      const defs = svgDoc.querySelector('defs');
      if (!defs) return;

      const lines = svgDoc.querySelectorAll('line');
      const uniqueColors = new Set<string>();
      for (const line of lines) {
        const stroke = line.getAttribute('stroke');
        if (stroke) uniqueColors.add(stroke);
      }

      if (uniqueColors.size === 0) return;

      const arrowheadMarker = defs.querySelector('#arrowhead');
      const sourceDotMarker = defs.querySelector('#source-dot');

      for (const color of uniqueColors) {
        const colorSuffix = color.replace('#', '');

        if (arrowheadMarker) {
          const cloned = arrowheadMarker.cloneNode(true) as Element;
          cloned.setAttribute('id', `arrowhead-${colorSuffix}`);
          const path = cloned.querySelector('path');
          if (path) path.setAttribute('fill', color);
          defs.appendChild(cloned);
        }

        if (sourceDotMarker) {
          const cloned = sourceDotMarker.cloneNode(true) as Element;
          cloned.setAttribute('id', `source-dot-${colorSuffix}`);
          const circle = cloned.querySelector('circle');
          if (circle) circle.setAttribute('fill', color);
          defs.appendChild(cloned);
        }
      }

      for (const line of lines) {
        const stroke = line.getAttribute('stroke');
        if (!stroke) continue;
        const colorSuffix = stroke.replace('#', '');

        const markerEnd = line.getAttribute('marker-end');
        if (markerEnd && markerEnd.includes('arrowhead')) {
          line.setAttribute('marker-end', `url(#arrowhead-${colorSuffix})`);
        }

        const markerStart = line.getAttribute('marker-start');
        if (markerStart && markerStart.includes('source-dot')) {
          line.setAttribute('marker-start', `url(#source-dot-${colorSuffix})`);
        }
      }
    }
    ```

    **`ensureSvgDimensions(svgDoc: Document): void`** (DOM-based, addresses Codex concern about DOM manipulation):
    Checks the root `<svg>` element for `width` and `height` attributes. If missing (the live DiagramRenderer only has `viewBox` and `style="width: 100%; height: auto;"`), adds `width="1200"` and `height="800"`. Also removes the `style` attribute containing percentage-based sizing since it's not meaningful for standalone SVG export.

    ```typescript
    export function ensureSvgDimensions(svgDoc: Document): void {
      const svgRoot = svgDoc.documentElement;
      if (!svgRoot.hasAttribute('width')) {
        svgRoot.setAttribute('width', '1200');
      }
      if (!svgRoot.hasAttribute('height')) {
        svgRoot.setAttribute('height', '800');
      }
      // Remove percentage-based style (not meaningful in exported SVG)
      const style = svgRoot.getAttribute('style');
      if (style && style.includes('width: 100%')) {
        svgRoot.removeAttribute('style');
      }
    }
    ```

    **`async exportDiagram(svgEl: SVGSVGElement, options: ExportOptions): Promise<void>`** (main entry point):

    The main pipeline:
    1. **Re-entry guard:** If `exportInFlight` is true, return silently.
    2. **Clone SVG to Document:** `const serialized = new XMLSerializer().serializeToString(svgEl);` then `const parser = new DOMParser(); const svgDoc = parser.parseFromString(serialized, 'image/svg+xml');`
    3. **Validate logo data URLs (EXPO-03 assertion):** Call `validateImageDataUrls(svgDoc)`. If false, log a warning (`console.warn('Some images are not base64 data URLs -- export may have missing logos')`) but continue.
    4. **Ensure dimensions:** Call `ensureSvgDimensions(svgDoc)`
    5. **Inject font** (with try/catch -- font failure is non-blocking per error handling contract):
       ```typescript
       try {
         const fontBase64 = await loadFontBase64();
         injectFont(svgDoc, fontBase64);
       } catch {
         console.warn('Font injection failed, exporting without embedded font');
       }
       ```
    6. **Fix context-stroke markers:** Call `fixContextStroke(svgDoc)`
    7. **Serialize back to string:** `const finalSvg = new XMLSerializer().serializeToString(svgDoc.documentElement);`
    8. **Generate filename:** `const filename = generateFilename(options.customerName, options.title, options.format);`
    9. **Branch on format:**
       - `'svg'`: call `downloadSvg(finalSvg, filename)`
       - `'png'`: call `await downloadPng(finalSvg, filename)`
